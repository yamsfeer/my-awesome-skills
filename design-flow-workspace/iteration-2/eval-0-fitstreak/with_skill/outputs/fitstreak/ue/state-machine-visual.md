# FitStreak 可视化状态机

## Part A：Mermaid 状态流程图

```mermaid
stateDiagram-v2
  [*] --> SplashPage

  state "启动页" as SplashPage {
    [*] --> splash_checking
    splash_checking --> [*] : token_valid → 跳首页
    splash_checking --> [*] : token_invalid → 跳登录页
  }

  SplashPage --> LoginPage : token 无效
  SplashPage --> HomePage : token 有效

  state "登录页" as LoginPage {
    [*] --> login_idle
    login_idle --> login_code_sending : 点击获取验证码
    login_code_sending --> login_code_input : 发送成功
    login_code_sending --> login_idle : 发送失败（Toast提示）
    login_code_input --> login_verifying : 输入满6位自动触发
    login_code_input --> login_code_sending : 点击重新获取（倒计时结束后）
    login_code_input --> login_idle : 更换手机号
    login_verifying --> [*] : 验证成功 → 跳首页
    login_verifying --> login_code_input : 验证码错误/过期
  }

  LoginPage --> HomePage : 验证通过

  state "首页" as HomePage {
    [*] --> home_loading
    home_loading --> home_success_not_checked : 加载成功，今日未打卡
    home_loading --> home_success_checked : 加载成功，今日已打卡
    home_loading --> home_error : 加载失败
    home_success_not_checked --> home_loading : 下拉刷新
    home_success_not_checked --> home_streak_broken_dialog : 检测到Streak断了
    home_streak_broken_dialog --> home_success_not_checked : 点击继续加油
    home_success_checked --> home_loading : 下拉刷新
    home_error --> home_loading : 点击重试
  }

  HomePage --> CheckInPage : 点击今日打卡（未打卡状态）
  HomePage --> ProfilePage : 点击好友头像

  state "打卡页" as CheckInPage {
    [*] --> checkin_loading
    checkin_loading --> checkin_idle : 今日未打卡
    checkin_loading --> checkin_already_checked : 今日已打卡
    checkin_idle --> checkin_submitting : 点击完成打卡
    checkin_submitting --> checkin_success_animation : 提交成功
    checkin_submitting --> checkin_already_checked : 409 今日已打卡
    checkin_submitting --> checkin_idle : 其他错误
    checkin_success_animation --> [*] : 3秒后/点击好的 → 返回首页
    checkin_already_checked --> [*] : 点击返回首页
  }

  CheckInPage --> HomePage : 打卡成功返回

  state "社交页" as SocialPage {
    [*] --> social_loading
    social_loading --> social_success : 有好友
    social_loading --> social_empty_friends : 无好友
    social_loading --> social_error : 加载失败
    social_success --> social_loading : 切换本周/本月
    social_error --> social_loading : 点击重试
  }

  SocialPage --> ProfilePage : 点击好友
  SocialPage --> AddFriendPage : 点击添加好友

  state "个人主页" as ProfilePage {
    [*] --> profile_loading
    profile_loading --> profile_success_self : 查看自己
    profile_loading --> profile_success_other : 查看他人
    profile_loading --> profile_error : 加载失败
    profile_error --> profile_loading : 点击重试
  }

  state "运动计划页" as PlanPage {
    [*] --> plan_loading
    plan_loading --> plan_success_has_plan : 有激活计划
    plan_loading --> plan_success_no_plan : 无激活计划
    plan_loading --> plan_error : 加载失败
    plan_error --> plan_loading : 点击重试
  }

  note right of CheckInPage
    🔴 每日仅可打卡一次
    🔴 提交时禁止重复点击
  end note

  note right of LoginPage
    🔴 手机号验证码限流
    🔴 验证码60秒倒计时
  end note
```
