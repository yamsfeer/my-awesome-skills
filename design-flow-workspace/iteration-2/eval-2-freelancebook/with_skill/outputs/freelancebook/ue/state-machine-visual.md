# FreelanceBook — 可视化状态机

## Part A：Mermaid 状态流程图

```mermaid
stateDiagram-v2
  [*] --> Auth

  state "登录/注册页" as Auth {
    [*] --> auth_idle
    auth_idle --> auth_logging_in : 提交登录
    auth_idle --> auth_registering : 提交注册
    auth_logging_in --> Dashboard : 登录成功
    auth_logging_in --> auth_login_error : 账号密码错误
    auth_login_error --> auth_idle : 重新输入
    auth_registering --> Dashboard : 注册成功
    auth_registering --> auth_email_error : 邮箱已存在
    auth_email_error --> auth_idle : 修改邮箱
  }

  Auth --> Dashboard : 认证通过

  state "仪表板" as Dashboard {
    [*] --> dash_loading
    dash_loading --> dash_success : 有数据
    dash_loading --> dash_empty : 新用户无数据
    dash_loading --> dash_error : 加载失败
    dash_success --> dash_loading : 下拉刷新
    dash_error --> dash_loading : 点击重试
  }

  state "客户列表" as ClientList {
    [*] --> cl_loading
    cl_loading --> cl_success : items > 0
    cl_loading --> cl_empty : items = 0
    cl_loading --> cl_error : 加载失败
    cl_success --> cl_searching : 输入搜索
    cl_searching --> cl_success : 搜索有结果
    cl_searching --> cl_search_empty : 搜索无结果
    cl_search_empty --> cl_success : 清除搜索
    cl_error --> cl_loading : 重试
  }

  state "客户详情" as ClientDetail {
    [*] --> cd_loading
    cd_loading --> cd_success : 加载成功
    cd_loading --> cd_not_found : 404
    cd_loading --> cd_error : 加载失败
    cd_success --> cd_archive_confirm : 点击归档
    cd_archive_confirm --> cd_archiving : 确认归档
    cd_archive_confirm --> cd_success : 取消
    cd_archiving --> cd_success : 归档完成
    cd_error --> cd_loading : 重试
  }

  state "发票列表" as InvoiceList {
    [*] --> il_loading
    il_loading --> il_success : items > 0
    il_loading --> il_empty : items = 0
    il_loading --> il_error : 加载失败
    il_success --> il_loading_more : 滚动到底
    il_loading_more --> il_success : 加载完成
    il_error --> il_loading : 重试
  }

  state "发票详情" as InvoiceDetail {
    [*] --> id_loading
    id_loading --> id_success : 加载成功
    id_loading --> id_not_found : 404
    id_loading --> id_error : 加载失败
    id_success --> id_updating_status : 更新状态
    id_updating_status --> id_success : 更新完成
    id_success --> id_payment_dialog : 记录收款
    id_payment_dialog --> id_saving_payment : 确认收款
    id_payment_dialog --> id_success : 取消
    id_saving_payment --> id_success : 保存成功
    id_saving_payment --> id_payment_dialog : 保存失败
    id_error --> id_loading : 重试
  }

  state "收入统计" as Stats {
    [*] --> st_loading
    st_loading --> st_success : 有数据
    st_loading --> st_empty : 无数据
    st_loading --> st_error : 加载失败
    st_success --> st_loading : 切换年度/周期
    st_error --> st_loading : 重试
  }

  state "税务预估" as Tax {
    [*] --> tax_loading
    tax_loading --> tax_success : 加载成功
    tax_loading --> tax_error : 加载失败
    tax_success --> tax_recalculating : 修改扣除项
    tax_recalculating --> tax_success : 重算完成
    tax_success --> tax_saving : 保存配置
    tax_saving --> tax_success : 保存完成
    tax_error --> tax_loading : 重试
  }

  Dashboard --> ClientList : 导航到客户
  Dashboard --> InvoiceList : 导航到发票
  Dashboard --> Stats : 导航到统计
  ClientList --> ClientDetail : 点击客户
  ClientDetail --> InvoiceDetail : 点击发票
  InvoiceList --> InvoiceDetail : 点击发票行
  Dashboard --> Tax : 导航到税务

  note right of Auth
    🔴 未登录用户唯一入口
    注册自动开启30天试用
  end note

  note right of InvoiceDetail
    🔴 发票状态流：
    draft→sent→paid/overdue
    状态不可逆转
  end note
```
